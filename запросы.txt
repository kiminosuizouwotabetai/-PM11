select tasks.id AS task_id, tasks.title as task_title, users.username, categories.name as category_name from tasks
join users on tasks.user_id = users.id
join categories ON tasks.category_id = categories.id;

---

select categories.name as category_name, count(tasks.id) as task_count from categories
join tasks on categories.id = tasks.category_id
group by categories.id, categories.name;

---

select tasks.title, tasks.due_date, priorities.name as priority_name from tasks
join priorities on tasks.priority_id = priorities.id
where priorities.level = 3 and tasks.is_completed = false;

---

select tc.comment_text, users.username as author, tasks.title as task_title, tc.created_at from task_comments tc
join users on tc.user_id = users.id
join tasks on tc.task_id = tasks.id;

---

select tasks.id, tasks.title, count(tc.id) as comments_count from tasks
join task_comments tc on tasks.id = tc.task_id
group by tasks.id, tasks.title
order by comments_count desc;
